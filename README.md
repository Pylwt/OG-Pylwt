function zeezyFrontflip(act,inp,obj)
	if inp == Enum.UserInputState.Begin then
		zeezy.Character.Humanoid:ChangeState("Jumping")

		for i = 1,360 do

			end)
		end
		wait(0.55)
		zeezy.Character.Humanoid.Sit = false
